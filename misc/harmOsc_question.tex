\begin{eq}
	\bra{\Phi}V\ket{\Phi'}
	=
	\sum_{nn'} \braket{\phi}{n} \bra{n}V\ket{n'}\braket{n'}{\Phi'}
	=
	\sum_n \Phi_n \sum_{n'} \Phi_{n'} V_{n n'}
\end{eq}
where
\begin{eq}
	V_{nn'} = \fint{r}r^2 R_{nl}(r) V R_{n' l}(r)
\end{eq}
